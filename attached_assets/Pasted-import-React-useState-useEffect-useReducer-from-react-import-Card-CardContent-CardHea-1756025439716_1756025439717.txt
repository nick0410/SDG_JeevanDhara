// Reducer for state management
const appReducer = (state, action) => {
  switch (action.type) {
    case 'CONNECT_WALLET':
      return { ...state, wallet: action.payload };
    case 'SET_USER':
      return { ...state, user: action.payload };
    case 'SELECT_SDG':
      return { ...state, selectedSDG: action.payload };
    case 'SET_LANGUAGE':
      return { ...state, language: action.payload };
    case 'SET_OFFLINE':
      return { ...state, isOffline: action.payload };
    case 'ADD_NOTIFICATION':
      return { ...state, notifications: [...state.notifications, action.payload] };
    case 'UPDATE_ML_RECOMMENDATIONS':
      return { ...state, mlRecommendations: action.payload };
    case 'ADD_BLOCKCHAIN_TX':
      return { ...state, blockchainTransactions: [...state.blockchainTransactions, action.payload] };
    default:
      return state;
  }
};
